package Controlador;
import com.mysql.jdbc.PreparedStatement;
/**
 *
 * @author kevin
 */
public class registrarPersonal extends Conexion {
      public boolean registrar(String nombres, String ap, String am, String sexo,
              int edad, String ocupacion, String nacion, String religion, String calle, int numero, 
              String colonia, String municipio,
              String estado, int telefono, String email, int segurosocial, String discapacidades, 
              String enfermedadescronicas){
      PreparedStatement pst=null;
      try{
        String consulta="insert into personal values(?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)";
          pst=(PreparedStatement) getconexionn().prepareStatement(consulta);
        
          
          pst.setString(1,nombres);
          pst.setString(2, ap);
          pst.setString(3, am);
          pst.setString(4, sexo);
          pst.setInt(5,edad);
          pst.setString(6, ocupacion);
          pst.setString(7, nacion);
          pst.setString(8, religion);
          pst.setString(9, calle);
          pst.setInt(10, numero);
          pst.setString(11, colonia);
          pst.setString(12, municipio);
          pst.setString(13, estado);
          pst.setInt(14, telefono);
          pst.setString(15, email);
          pst.setInt(16, segurosocial);
            pst.setString(17, discapacidades);  
            pst.setString(18, enfermedadescronicas);
            pst.setString(19,"0");
            
        
         
          if (pst.executeUpdate()==1) {
              return true;
          }  
      }catch(Exception e){
          System.out.println("error:"+e);
      }finally{
        try{
              if (getconexionn()!=null)getconexionn().close();
                if(pst!=null)pst.close();;
        }catch(Exception e){
            
        }
      }
      return false;
  }
}
